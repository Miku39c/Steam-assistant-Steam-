REM 提取最后面的内容
REM 查找并获取行数的字符串，然后存储到文件里
findstr /n  "(async()=>{" ..\main.js >> temp.txt
REM 查找并提取字符串中的数字 然后存储到文件里
for /f "delims=^: tokens=1,*" %i in (temp.txt) do (echo %i >temp1.txt)
REM 从文件读取行数到变量
for /f %x in ('more temp1.txt') do (set i=%x)
REM 转数字(可能)
set /a start = i-1
more +%start% "..\main.js" >> Release.js
del temp.txt
del temp1.txt
